val mk_with_id : Base.int -> Base.string -> (Base.int -> 'a) -> 'a

mk_with_id i s f applies f to i is i < 0, otherwise creates a fresh id, reigsters the name s to that fresh id, and returns the result of applying f to that fresh id.
